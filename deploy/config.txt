# PiTrex baremetal boot configuration for vekterm.
#
# vekterm boots straight into a baremetal kernel — no operating system. Two
# kernels ship on the card because the peripheral base address and CPU arch
# differ per Pi family and are fixed at compile time:
#
#   kernel.img   Pi Zero / Zero W   (BCM2835, ARMv6, peripherals @ 0x20000000)
#   kernel7.img  Pi Zero 2 W / 2/3  (BCM2837, ARMv7, peripherals @ 0x3F000000)
#
# The board-conditional filters below make the firmware load the right one.
# (These are also the firmware's default kernel names per SoC, so the selection
# is belt-and-braces — but explicit is clearer and matches the official PiTrex
# distribution's scheme.)
[pi0]
kernel=kernel.img
[pi02]
kernel=kernel7.img

[all]
# GPIO setup from gtoal/pitrex's README-BAREMETAL: stop the firmware bringing
# pins up in a mode that could damage the VIA interface.
gpio=0-5,16-24,26-29=ip
gpio=6-13,25=op
gpio=24=np

# NOTE: do NOT enable dtoverlay here. An overlay makes the firmware load a
# device tree + .dtbo files from /overlays/, which this baremetal card does not
# carry; on some firmware that stops the kernel from booting at all. The
# official PiTrex distribution ships this line commented out, so we do too.
#dtoverlay=dwc2,dr_mode=host

# Clocking matches the official PiTrex baremetal distribution (proven to boot on
# both the Pi Zero and the Zero 2 W): enable the mini-UART on the header, pin the
# core clock at 250 MHz, and force turbo so it can't scale. The baremetal runtime
# is built to compute the mini-UART divisor from this same 250 MHz core clock
# (see VT_UART_CLOCK / bareMetalMain.c), so the serial baud comes out correct.
enable_uart=1
core_freq=250
force_turbo=1

# Boot the 32-bit (ARMv6/ARMv7) kernels, not a 64-bit one.
arm_64bit=0
